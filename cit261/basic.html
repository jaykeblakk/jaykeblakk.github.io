Objects, Inheritance, & Methods

    
    function Computer(drive, processor, os, graphics)
    {
        this.drive = drive;
        this.processor = processor;
        this.os = os;
        this.graphics = graphics;
    }

    Computer.prototype.specs = function() {
        var display = document.getElementById('display5');
        display.innerHTML += "This computer is a desktop that currently has these specs: ";
        display.innerHTML += "Hard Drive - " + this.drive;
        display.innerHTML += "Processor - " + this.processor;
        display.innerHTML += "OS - " + this.os;
        display.innerHTML += "Graphics - " + this.graphics;
    }

    function Laptop(drive, processor, os, graphics, brand)
    {
        Computer.call(this, drive, processor, os, graphics);
        this.brand = brand;
    }

    Laptop.prototype.specs = function()
    {
        var display = document.getElementById('display5');
        display.innerHTML += "This computer is a laptop that currently has these specs: ";
        display.innerHTML += "Hard Drive - " + this.drive;
        display.innerHTML += "Processor - " + this.processor;
        display.innerHTML += "OS - " + this.os;
        display.innerHTML += "Graphics - " + this.graphics;
        display.innerHTML += "Brand - " + this.brand;
    }

    function createComputers()
    {
        var laptop1 = new Laptop('Seagate', 'AMD', 'Windows 7', 'AMD', 'Dell');
        var computer1 = new Computer('Western Digital', 'Intel', 'Windows 10', 'NVIDIA');
        var laptop2 = new Laptop('Barracuda', 'Apple A5', 'OS X', 'On-Board graphics', 'Apple Macbook Pro')
        computer1.specs();
        laptop1.specs();
        laptop2.specs();
    }

    function loadJSON() {
        var xhttp = new XMLHttpRequest();
        xhttp.onreadystatechange = function() {
            if (this.readyState == 4 && this.status == 200) {
                document.getElementById("display6").innerHTML += this.responseText;
            }
        };
        xhttp.open("GET", "ajaxtest.json", true);
        xhttp.send();
    }

DOM Manipulation

    
    function changeDocument() {
        var header = document.getElementById("changeheader");
        var para = document.createElement("p");
        var text = document.createTextNode("DOM Manipulation Code Success!");
        var titles = document.querySelectorAll('h1');
        titles[7].style.color = "black";
        para.appendChild(text);
        header.appendChild(para);
    }

    function fixDocument() {
        var header = document.getElementById("changeheader");
        header.removeChild(header.childNodes[1]);
        header.style.color = "white";
    }
